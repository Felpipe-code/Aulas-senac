SELECT personagens.nome, desenhos.nome
FROM personagens
JOIN desenhos ON personagens.desenho_id = desenhos.id
ORDER BY desenhos.nome ASC;


SELECT albuns.titulo, artistas.nome
FROM albuns
JOIN artistas ON albuns.artista_id = artistas.id
ORDER BY albuns.ano_lancamento DESC;


SELECT livros.titulo, autores.nome
FROM livros
JOIN autores ON livros.autor_id = autores.id

SELECT desenhos.nome, COUNT(personagens.id) AS total_personagens
FROM desenhos
LEFT JOIN personagens ON personagens.desenho_id = desenhos.id
GROUP BY desenhos.nome;


SELECT desenhos.nome, AVG(personagens.idade) AS idade_media
FROM desenhos
JOIN personagens ON personagens.desenho_id = desenhos.id
GROUP BY desenhos.nome;
